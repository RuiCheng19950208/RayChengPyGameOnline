# Tank Game 环境配置文件模板
# 将此文件复制为 .env 并根据需要修改配置

# 游戏基础配置
SCREEN_WIDTH=800
SCREEN_HEIGHT=600
FPS=60
TANK_SPEED=300

# 服务器网络配置
# 服务器模式：设置为 0.0.0.0 监听所有网络接口，允许其他电脑连接
# 客户端模式：设置为服务器电脑的实际IP地址
SERVER_HOST=localhost
SERVER_PORT=8765
MAX_PLAYERS_PER_ROOM=8

# 子弹配置
BULLET_SPEED=300
BULLET_DAMAGE=25
BULLET_LIFETIME=5.0

# 字体配置
# 如果字体文件不存在，会自动使用默认字体
DEFAULT_FONT_PATH=assets/STHeiti Light.ttc

# ====================================
# 多台电脑联机配置示例
# ====================================
# 
# 🖥️  服务器电脑配置：
# SERVER_HOST=0.0.0.0
# SERVER_PORT=8765
# 
# 💻 客户端电脑配置（假设服务器IP为192.168.1.100）：
# SERVER_HOST=192.168.1.100
# SERVER_PORT=8765
#
# ====================================
# 使用方法
# ====================================
#
# 1. 将此文件复制为 .env：
#    cp env_template.txt .env
#
# 2. 编辑 .env 文件，修改 SERVER_HOST 和其他配置
#
# 3. 启动游戏：
#    服务器：python server/tank_game_server.py
#    客户端：python home/tank_game_client.py
#
# 4. 或使用命令行参数：
#    python home/tank_game_client.py --host 192.168.1.100
#    python home/tank_game_client.py --server ws://192.168.1.100:8765
